```python
def _gutter_width(self) -> int:
    if not self.config.show_line_numbers:
        return 0
    return max(3, len(str(len(self.buffer.lines))) + 1)

def _text_origin_x(self) -> int:
    return self._gutter_width()

def _cursor_screen_x(self) -> int:
    prefix = self.buffer.current_line()[: self.buffer.cursor_col]
    return self._text_origin_x() + display_width(prefix) - self.col_offset
```
